    defmodule User do
  use Ecto.Schema
  use Ecto.Model

  schema "users" do
    field :email, :string
    field :iban, :string
  end

  def changeset(user, params \\ :empty) do
    user
    |> cast(params, ~w(iban email), ~w())
    |> ExIban.EctoValidator.validate_iban(:iban)
  end
